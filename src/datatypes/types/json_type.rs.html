51#[derive(Debug, Clone, PartialEq, Eq, Hash, PartialOrd, Ord, Serialize, Deserialize)]
52pub enum JsonNativeType {
53    Null,
54    Bool,
55    Number(JsonNumberType),
56    String,
57    Array(Box<JsonNativeType>),
58    Object(JsonObjectType),
59}

61impl JsonNativeType {
62    pub fn u64() -> Self {
63        Self::Number(JsonNumberType::U64)
64    }
65
66    pub fn i64() -> Self {
67        Self::Number(JsonNumberType::I64)
68    }
69
70    pub fn f64() -> Self {
71        Self::Number(JsonNumberType::F64)
72    }
73}

75impl From<&JsonNativeType> for ConcreteDataType {
76    fn from(value: &JsonNativeType) -> Self {
77        match value {
78            JsonNativeType::Null => ConcreteDataType::null_datatype(),
79            JsonNativeType::Bool => ConcreteDataType::boolean_datatype(),
80            JsonNativeType::Number(JsonNumberType::U64) => ConcreteDataType::uint64_datatype(),
81            JsonNativeType::Number(JsonNumberType::I64) => ConcreteDataType::int64_datatype(),
82            JsonNativeType::Number(JsonNumberType::F64) => ConcreteDataType::float64_datatype(),
83            JsonNativeType::String => ConcreteDataType::string_datatype(),
84            JsonNativeType::Array(item_type) => {
85                ConcreteDataType::List(ListType::new(Arc::new(item_type.as_ref().into())))
86            }
87            JsonNativeType::Object(object) => {
88                let fields = object
89                    .iter()
90                    .map(|(type_name, field_type)| {
91                        StructField::new(type_name.clone(), field_type.into(), true)
92                    })
93                    .collect();
94                ConcreteDataType::Struct(StructType::new(Arc::new(fields)))
95            }
96        }
97    }
98}

100impl From<&ConcreteDataType> for JsonNativeType {
101    fn from(value: &ConcreteDataType) -> Self {
102        match value {
103            ConcreteDataType::Null(_) => JsonNativeType::Null,
104            ConcreteDataType::Boolean(_) => JsonNativeType::Bool,
105            ConcreteDataType::UInt64(_)
106            | ConcreteDataType::UInt32(_)
107            | ConcreteDataType::UInt16(_)
108            | ConcreteDataType::UInt8(_) => JsonNativeType::u64(),
109            ConcreteDataType::Int64(_)
110            | ConcreteDataType::Int32(_)
111            | ConcreteDataType::Int16(_)
112            | ConcreteDataType::Int8(_) => JsonNativeType::i64(),
113            ConcreteDataType::Float64(_) | ConcreteDataType::Float32(_) => JsonNativeType::f64(),
114            ConcreteDataType::String(_) => JsonNativeType::String,
115            ConcreteDataType::List(list_type) => {
116                JsonNativeType::Array(Box::new(list_type.item_type().into()))
117            }
118            ConcreteDataType::Struct(struct_type) => JsonNativeType::Object(
119                struct_type
120                    .fields()
121                    .iter()
122                    .map(|field| (field.name().to_string(), field.data_type().into()))
123                    .collect(),
124            ),
125            ConcreteDataType::Json(json_type) => json_type.native_type().clone(),
126            _ => unreachable!(),
127        }
128    }
129}

131impl Display for JsonNativeType {
132    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
133        match self {
134            JsonNativeType::Null => write!(f, "Null"),
135            JsonNativeType::Bool => write!(f, "Bool"),
136            JsonNativeType::Number(t) => {
137                write!(f, "Number({t:?})")
138            }
139            JsonNativeType::String => write!(f, "String"),
140            JsonNativeType::Array(item_type) => {
141                write!(f, "Array[{}]", item_type)
142            }
143            JsonNativeType::Object(object) => {
144                write!(
145                    f,
146                    "Object{{{}}}",
147                    object
148                        .iter()
149                        .map(|(k, v)| format!(r#""{k}": {v}"#))
150                        .collect::<Vec<_>>()
151                        .join(", ")
152                )
153            }
154        }
155    }
156}

158#[derive(Debug, Clone, PartialEq, Eq, Hash, PartialOrd, Ord, Serialize, Deserialize, Default)]
159pub enum JsonFormat {
160    #[default]
161    Jsonb,
162    Native(Box<JsonNativeType>),
163}

165/// JsonType is a data type for JSON data. It is stored as binary data of jsonb format.
166/// It utilizes current binary value and vector implementation.
167#[derive(Debug, Default, Clone, PartialEq, Eq, Hash, PartialOrd, Ord, Serialize, Deserialize)]
168pub struct JsonType {
169    pub format: JsonFormat,
170}
171
172impl JsonType {
173    pub fn new(format: JsonFormat) -> Self {
174        Self { format }
175    }
176
177    pub(crate) fn new_native(native: JsonNativeType) -> Self {
178        Self {
179            format: JsonFormat::Native(Box::new(native)),
180        }
181    }
182
183    pub(crate) fn native_type(&self) -> &JsonNativeType {
184        match &self.format {
185            JsonFormat::Jsonb => &JsonNativeType::String,
186            JsonFormat::Native(x) => x.as_ref(),
187        }
188    }
189
190    pub(crate) fn empty() -> Self {
191        Self {
192            format: JsonFormat::Native(Box::new(JsonNativeType::Null)),
193        }
194    }
195
196    /// Make json type a struct type, by:
197    /// - if the json is an object, its entries are mapped to struct fields, obviously;
198    /// - if not, the json is one of bool, number, string or array, make it a special field
199    ///   (see [plain_json_struct_type]).
200    pub(crate) fn as_struct_type(&self) -> StructType {
201        match &self.format {
202            JsonFormat::Jsonb => StructType::default(),
203            JsonFormat::Native(inner) => match ConcreteDataType::from(inner.as_ref()) {
204                ConcreteDataType::Struct(t) => t.clone(),
205                x => plain_json_struct_type(x),
206            },
207        }
208    }
209
210    /// Try to merge this json type with others, error on datatype conflict.
211    pub(crate) fn merge(&mut self, other: &JsonType) -> Result<()> {
212        match (&self.format, &other.format) {
213            (JsonFormat::Jsonb, JsonFormat::Jsonb) => Ok(()),
214            (JsonFormat::Native(this), JsonFormat::Native(that)) => {
215                let merged = merge(this.as_ref(), that.as_ref())?;
216                self.format = JsonFormat::Native(Box::new(merged));
217                Ok(())
218            }
219            _ => MergeJsonDatatypeSnafu {
220                reason: "json format not match",
221            }
222            .fail(),
223        }
224    }
225
226    pub(crate) fn is_mergeable(&self, other: &JsonType) -> bool {
227        match (&self.format, &other.format) {
228            (JsonFormat::Jsonb, JsonFormat::Jsonb) => true,
229            (JsonFormat::Native(this), JsonFormat::Native(that)) => {
230                is_mergeable(this.as_ref(), that.as_ref())
231            }
232            _ => false,
233        }
234    }
235}

237/// A special struct type for denoting "plain"(not object) json value. It has only one field, with
238/// fixed name [JSON_PLAIN_FIELD_NAME] and with metadata [JSON_PLAIN_FIELD_METADATA_KEY] = `"true"`.
239pub(crate) fn plain_json_struct_type(item_type: ConcreteDataType) -> StructType {
240    let mut field = StructField::new(JSON_PLAIN_FIELD_NAME.to_string(), item_type, true);
241    field.insert_metadata(JSON_PLAIN_FIELD_METADATA_KEY, true);
242    StructType::new(Arc::new(vec![field]))
243}

245fn is_mergeable(this: &JsonNativeType, that: &JsonNativeType) -> bool {
246    fn is_mergeable_object(this: &JsonObjectType, that: &JsonObjectType) -> bool {
247        for (type_name, that_type) in that {
248            if let Some(this_type) = this.get(type_name)
249                && !is_mergeable(this_type, that_type)
250            {
251                return false;
252            }
253        }
254        true
255    }
256
257    match (this, that) {
258        (this, that) if this == that => true,
259        (JsonNativeType::Array(this), JsonNativeType::Array(that)) => {
260            is_mergeable(this.as_ref(), that.as_ref())
261        }
262        (JsonNativeType::Object(this), JsonNativeType::Object(that)) => {
263            is_mergeable_object(this, that)
264        }
265        (JsonNativeType::Null, _) | (_, JsonNativeType::Null) => true,
266        _ => false,
267    }
268}
269
270fn merge(this: &JsonNativeType, that: &JsonNativeType) -> Result<JsonNativeType> {
271    fn merge_object(this: &JsonObjectType, that: &JsonObjectType) -> Result<JsonObjectType> {
272        let mut this = this.clone();
273        // merge "that" into "this" directly:
274        for (type_name, that_type) in that {
275            if let Some(this_type) = this.get_mut(type_name) {
276                let merged_type = merge(this_type, that_type)?;
277                *this_type = merged_type;
278            } else {
279                this.insert(type_name.clone(), that_type.clone());
280            }
281        }
282        Ok(this)
283    }
284
285    match (this, that) {
286        (this, that) if this == that => Ok(this.clone()),
287        (JsonNativeType::Array(this), JsonNativeType::Array(that)) => {
288            merge(this.as_ref(), that.as_ref()).map(|x| JsonNativeType::Array(Box::new(x)))
289        }
290        (JsonNativeType::Object(this), JsonNativeType::Object(that)) => {
291            merge_object(this, that).map(JsonNativeType::Object)
292        }
293        (JsonNativeType::Null, x) | (x, JsonNativeType::Null) => Ok(x.clone()),
294        _ => MergeJsonDatatypeSnafu {
295            reason: format!("datatypes have conflict, this: {this}, that: {that}"),
296        }
297        .fail(),
298    }
299}

301impl DataType for JsonType {
302    fn name(&self) -> String {
303        match &self.format {
304            JsonFormat::Jsonb => JSON_TYPE_NAME.to_string(),
305            JsonFormat::Native(x) => format!("Json<{x}>"),
306        }
307    }
308
309    fn logical_type_id(&self) -> LogicalTypeId {
310        LogicalTypeId::Json
311    }
312
313    fn default_value(&self) -> Value {
314        Bytes::default().into()
315    }
316
317    fn as_arrow_type(&self) -> ArrowDataType {
318        match self.format {
319            JsonFormat::Jsonb => ArrowDataType::Binary,
320            JsonFormat::Native(_) => ArrowDataType::Struct(Fields::empty()),
321        }
322    }
323
324    fn create_mutable_vector(&self, capacity: usize) -> Box<dyn MutableVector> {
325        match self.format {
326            JsonFormat::Jsonb => Box::new(BinaryVectorBuilder::with_capacity(capacity)),
327            JsonFormat::Native(_) => Box::new(JsonVectorBuilder::with_capacity(capacity)),
328        }
329    }
330
331    fn try_cast(&self, from: Value) -> Option<Value> {
332        match from {
333            Value::Binary(v) => Some(Value::Binary(v)),
334            _ => None,
335        }
336    }
337}

339/// Converts a json type value to string
340pub fn jsonb_to_string(val: &[u8]) -> Result<String> {
341    match jsonb::from_slice(val) {
342        Ok(jsonb_value) => {
343            let serialized = jsonb_value.to_string();
344            Ok(serialized)
345        }
346        Err(e) => InvalidJsonbSnafu { error: e }.fail(),
347    }
348}
349
350/// Converts a json type value to serde_json::Value
351pub fn jsonb_to_serde_json(val: &[u8]) -> Result<serde_json::Value> {
352    let json_string = jsonb_to_string(val)?;
353    serde_json::Value::from_str(json_string.as_str())
354        .context(DeserializeSnafu { json: json_string })
355}
356
357/// Parses a string to a json type value
358pub fn parse_string_to_jsonb(s: &str) -> Result<Vec<u8>> {
359    jsonb::parse_value(s.as_bytes())
360        .map_err(|_| InvalidJsonSnafu { value: s }.build())
361        .map(|json| json.to_vec())
362}

364#[cfg(test)]
365mod tests {
366    use super::*;
367    use crate::json::JsonStructureSettings;
368
369    #[test]
370    fn test_merge_json_type() -> Result<()> {
371        fn test(
372            json: &str,
373            json_type: &mut JsonType,
374            expected: std::result::Result<&str, &str>,
375        ) -> Result<()> {
376            let json: serde_json::Value = serde_json::from_str(json).unwrap();
377
378            let settings = JsonStructureSettings::Structured(None);
379            let value = settings.encode(json)?;
380            let value_type = value.data_type();
381            let Some(other) = value_type.as_json() else {
382                unreachable!()
383            };
384
385            let result = json_type.merge(other);
386            match (result, expected) {
387                (Ok(()), Ok(expected)) => {
388                    assert_eq!(json_type.name(), expected);
389                    assert!(json_type.is_mergeable(other));
390                }
391                (Err(err), Err(expected)) => {
392                    assert_eq!(err.to_string(), expected);
393                    assert!(!json_type.is_mergeable(other));
394                }
395                _ => unreachable!(),
396            }
397            Ok(())
398        }
399
400        let json_type = &mut JsonType::new_native(JsonNativeType::Null);
401
402        // can merge with json object:
403        let json = r#"{
404            "hello": "world",
405            "list": [1, 2, 3],
406            "object": {"a": 1}
407        }"#;
408        let expected = r#"Json<Object{"hello": String, "list": Array[Number(I64)], "object": Object{"a": Number(I64)}}>"#;
409        test(json, json_type, Ok(expected))?;
410
411        // cannot merge with other non-object json values:
412        let jsons = [r#""s""#, "1", "[1]"];
413        let expects = [
414            r#"Failed to merge JSON datatype: datatypes have conflict, this: Object{"hello": String, "list": Array[Number(I64)], "object": Object{"a": Number(I64)}}, that: String"#,
415            r#"Failed to merge JSON datatype: datatypes have conflict, this: Object{"hello": String, "list": Array[Number(I64)], "object": Object{"a": Number(I64)}}, that: Number(I64)"#,
416            r#"Failed to merge JSON datatype: datatypes have conflict, this: Object{"hello": String, "list": Array[Number(I64)], "object": Object{"a": Number(I64)}}, that: Array[Number(I64)]"#,
417        ];
418        for (json, expect) in jsons.into_iter().zip(expects.into_iter()) {
419            test(json, json_type, Err(expect))?;
420        }
421
422        // cannot merge with other json object with conflict field datatype:
423        let json = r#"{
424            "hello": 1,
425            "float": 0.123,
426            "no": 42
427        }"#;
428        let expected = r#"Failed to merge JSON datatype: datatypes have conflict, this: String, that: Number(I64)"#;
429        test(json, json_type, Err(expected))?;
430
431        // can merge with another json object:
432        let json = r#"{
433            "hello": "greptime",
434            "float": 0.123,
435            "int": 42
436        }"#;
437        let expected = r#"Json<Object{"float": Number(F64), "hello": String, "int": Number(I64), "list": Array[Number(I64)], "object": Object{"a": Number(I64)}}>"#;
438        test(json, json_type, Ok(expected))?;
439
440        // can merge with some complex nested json object:
441        let json = r#"{
442            "list": [4],
443            "object": {"foo": "bar", "l": ["x"], "o": {"key": "value"}},
444            "float": 0.456,
445            "int": 0
446        }"#;
447        let expected = r#"Json<Object{"float": Number(F64), "hello": String, "int": Number(I64), "list": Array[Number(I64)], "object": Object{"a": Number(I64), "foo": String, "l": Array[String], "o": Object{"key": String}}}>"#;
448        test(json, json_type, Ok(expected))?;
449
450        Ok(())
451    }
